\begin{frame}{Solution}

	Now,
  \begin{align}
 \vec{B - A}= \myvec{1 - (-1) \\ -2 - 2 \\ 5 - 1} = \myvec{2 \\ -4 \\ 4}, \\[6pt]
\vec{C - D} = \myvec{4 - 2 \\ -7 - (-3) \\ 8 - 4}= \myvec{2 \\ -4 \\ 4}.
 \end{align}

 Hence,
 \begin{align}
        \vec{B - A} = \vec{C - D}
 \end{align}
  $\implies$


 \begin{align}
        \vec{C-B} = \vec{D-A}
 \end{align}
	Therefore, $A, B, C, D$ are the vertices of a parallelogram.

\end{frame}
